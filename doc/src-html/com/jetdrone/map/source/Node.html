001package com.jetdrone.map.source;
002
003import java.io.Serializable;
004import java.util.HashMap;
005import java.util.Map;
006
007@SuppressWarnings("serial")
008public class Node implements Serializable {
009
010  private double lat;
011  private double lon;
012  private int layer;
013
014  private Map<String, String> tags;
015
016  public Node(double lat, double lon) {
017    this.lat = lat;
018    this.lon = lon;
019  }
020
021  public double getLat() {
022    return lat;
023  }
024
025  public double getLon() {
026    return lon;
027  }
028
029  public int getLayer() {
030    return layer;
031  }
032
033  public void setLayer(int layer) {
034    this.layer = layer;
035  }
036
037  public void insertTag(String key, String value) {
038    
039    if(tags == null) tags = new HashMap<String, String>();
040    tags.put(key, value);
041  }
042}
