001package com.jetdrone.map.source;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008
009import com.jetdrone.map.BoundingBox;
010
011@SuppressWarnings("serial")
012public class Way extends BoundingBox implements Serializable {
013
014  private final List<Node> nd;
015
016  private int layer;
017  private String name;
018  private Map<String, String> tags;
019
020  public Way() {
021    super(360.0, 360.0, -360.0, -360.0);
022    nd = new ArrayList<Node>();
023  }
024
025  public void setName(String name) {
026    this.name = name;
027  }
028
029  public String getName() {
030    return name;
031  }
032
033  public int getLayer() {
034    return layer;
035  }
036
037  public void setLayer(int layer) {
038    this.layer = layer;
039  }
040
041  public void insertTag(String key, String value) {
042    if(tags == null) tags = new HashMap<String, String>();
043    tags.put(key, value);
044  }
045
046  public void addWayNode(Node node) {
047    nd.add(node);
048    double lat = node.getLat();
049    double lon = node.getLon();
050    
051    if(lat < getMinLat()) setMinLat(lat);
052    if(lat > getMaxLat()) setMaxLat(lat);
053    if(lon < getMinLon()) setMinLon(lon);
054    if(lon > getMaxLon()) setMaxLon(lon);
055  }
056
057  public Map<String, String> getTags() {
058    return tags;
059  }
060
061  public List<Node> getWayNode() {
062    return nd;
063  }
064}
