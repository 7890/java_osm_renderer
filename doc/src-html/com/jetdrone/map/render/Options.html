001package com.jetdrone.map.render;
002
003public final class Options {
004  
005  public static final int MAP = 1;
006  public static final int TILE = 2;
007
008  /* render mode */
009  static int mode = MAP;
010
011  static int minlayer = 12;
012  static int maxlayer = 17;
013  static int zoom_level = 14;
014  static String cfgfn;
015  static String osmfn;
016  static String outdir = "tiles"; /* tiles output directory */
017
018  private Options() {
019    // singleton
020  }
021
022  public static boolean parse(String[] args) {
023    try {
024      for (int i = 0; i < args.length; i++) {
025        if ("-m".equals(args[i])) {
026          mode = MAP;
027          continue;
028        }
029        if ("-t".equals(args[i])) {
030          mode = TILE;
031          continue;
032        }
033        if ("--minlayer".equals(args[i])) {
034          minlayer = Integer.parseInt(args[++i]);
035          continue;
036        }
037        if ("--maxlayer".equals(args[i])) {
038          maxlayer = Integer.parseInt(args[++i]);
039          continue;
040        }
041        if ("--zoom".equals(args[i])) {
042          zoom_level = Integer.parseInt(args[++i]);
043          minlayer = zoom_level;
044          maxlayer = zoom_level;
045          continue;
046        }
047        if ("-o".equals(args[i])) {
048          outdir = args[++i];
049          continue;
050        }
051        cfgfn = args[i];
052        osmfn = args[++i];
053        break;
054      }
055    } catch (RuntimeException re) {
056      return false;
057    }
058
059    return !(cfgfn == null || osmfn == null);
060  }
061}
