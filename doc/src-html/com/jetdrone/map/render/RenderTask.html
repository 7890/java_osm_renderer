001package com.jetdrone.map.render;
002
003import java.io.IOException;
004
005import com.jetdrone.map.render.backend.Renderer;
006
007public class RenderTask implements Runnable {
008
009  private final Renderer renderer;
010  private final String filename;
011  private final int x, y, zoom_level;
012
013  @SuppressWarnings("boxing")
014  public RenderTask(Renderer renderer, int x, int y, int zoom_level) {
015    this.renderer = renderer;
016    this.filename = String.format("tiles/%d_%d.png", x, y);
017    this.x = x;
018    this.y = y;
019    this.zoom_level = zoom_level;
020  }
021
022  @Override
023  public void run() {
024    try {
025      renderer.drawTile(filename, x, y, zoom_level);
026    } catch (IOException e) {
027      System.err.println("Error: RenderTask[" + this + "] caused by [" + e + "]");
028    }
029  }
030}
